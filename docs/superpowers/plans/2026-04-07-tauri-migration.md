# Electron → Tauri v2 Migration Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Electron with Tauri v2, eliminating Chromium bundling and Node.js server dependency.

**Architecture:** Tauri's Rust core manages the Python daemon subprocess, serves all API data via `#[tauri::command]` functions backed by `rusqlite`, and serves media via Asset Protocol. The existing React+Vite frontend stays, with `api.ts` switching from `fetch()` to `invoke()`.

**Tech Stack:** Tauri v2, Rust, rusqlite, serde, React 19, Vite 6, TypeScript

use rusqlite::Connection;
use std::collections::HashMap;
use std::path::{Path, PathBuf};
use std::sync::Mutex;
use std::time::Instant;

pub struct AppDb {
    pub conn: Mutex<Connection>,
    pub data_dir: PathBuf,
    pub config_dir: PathBuf,
    mappings_cache: Mutex<Option<(HashMap<String, String>, Instant)>>,
}

const CACHE_TTL_SECS: u64 = 60;

impl AppDb {
    pub fn new(data_dir: &Path, config_dir: &Path) -> Result<Self, rusqlite::Error> {
        let db_path = data_dir.join("life.db");
        let conn = Connection::open(&db_path)?;
        conn.pragma_update(None, "journal_mode", "WAL")?;

        // Ensure memos table exists (write access needed)
        conn.execute_batch(
            "CREATE TABLE IF NOT EXISTS memos (
                date TEXT PRIMARY KEY,
                content TEXT NOT NULL DEFAULT '',
                updated_at TEXT NOT NULL DEFAULT (datetime('now'))
            )",
        )?;

        Ok(Self {
            conn: Mutex::new(conn),
            data_dir: data_dir.to_path_buf(),
            config_dir: config_dir.to_path_buf(),
            mappings_cache: Mutex::new(None),
        })
    }

    pub fn get_meta_category(&self, activity: &str) -> String {
        let mappings = self.get_activity_mappings();
        mappings
            .get(activity)
            .cloned()
            .unwrap_or_else(|| "other".to_string())
    }

    pub fn get_activity_mappings(&self) -> HashMap<String, String> {
        let mut cache = self.mappings_cache.lock().unwrap();
        if let Some((ref map, ref ts)) = *cache {
            if ts.elapsed().as_secs() < CACHE_TTL_SECS {
                return map.clone();
            }
        }

        let conn = self.conn.lock().unwrap();
        let mut stmt = conn
            .prepare("SELECT activity, meta_category FROM activity_mappings")
            .unwrap();
        let map: HashMap<String, String> = stmt
            .query_map([], |row| {
                Ok((row.get::<_, String>(0)?, row.get::<_, String>(1)?))
            })
            .unwrap()
            .filter_map(|r| r.ok())
            .collect();

        *cache = Some((map.clone(), Instant::now()));
        map
    }
}

use std::process::{Child, Command, Stdio};
use std::sync::Mutex;

pub struct DaemonProcess {
    pub child: Mutex<Option<Child>>,
}

impl DaemonProcess {
    pub fn new() -> Self {
        Self {
            child: Mutex::new(None),
        }
    }

    pub fn start(&self, python_bin: &str, config_dir: &str, daemon_src: &str, data_dir: &str) {
        let mut guard = self.child.lock().unwrap();
        if guard.is_some() {
            return; // already running
        }

        let result = Command::new(python_bin)
            .args(["-m", "daemon", "start"])
            .current_dir(config_dir)
            .env("PYTHONPATH", daemon_src)
            .env("DATA_DIR", data_dir)
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .spawn();

        match result {
            Ok(child) => {
                println!("[daemon] started (pid {})", child.id());
                *guard = Some(child);
            }
            Err(e) => {
                eprintln!("[daemon] failed to start: {}", e);
            }
        }
    }

    pub fn stop(&self) {
        let mut guard = self.child.lock().unwrap();
        if let Some(mut child) = guard.take() {
            println!("[daemon] stopping...");
            let _ = child.kill();
            let _ = child.wait();
            println!("[daemon] stopped");
        }
    }

    pub fn is_running(&self) -> bool {
        let mut guard = self.child.lock().unwrap();
        if let Some(ref mut child) = *guard {
            match child.try_wait() {
                Ok(Some(_)) => {
                    *guard = None;
                    false
                }
                Ok(None) => true,
                Err(_) => false,
            }
        } else {
            false
        }
    }
}

impl Drop for DaemonProcess {
    fn drop(&mut self) {
        self.stop();
    }
}

#[tauri::command]
pub fn get_stats(date: String, db: State<AppDb>) -> Result<DayStats, String> {
    let conn = db.conn.lock().map_err(|e| e.to_string())?;
    let start = format!("{date}T00:00:00");
    let end = format!("{date}T23:59:59");

    let frames: i64 = conn
        .query_row(
            "SELECT COUNT(*) FROM frames WHERE timestamp BETWEEN ?1 AND ?2",
            params![start, end],
            |row| row.get(0),
        )
        .unwrap_or(0);

    let events: i64 = conn
        .query_row(
            "SELECT COUNT(*) FROM events WHERE timestamp BETWEEN ?1 AND ?2",
            params![start, end],
            |row| row.get(0),
        )
        .unwrap_or(0);

    let summaries: i64 = conn
        .query_row(
            "SELECT COUNT(*) FROM summaries WHERE timestamp BETWEEN ?1 AND ?2",
            params![start, end],
            |row| row.get(0),
        )
        .unwrap_or(0);

    let avg_motion: f64 = conn
        .query_row(
            "SELECT COALESCE(AVG(motion_score), 0) FROM frames WHERE timestamp BETWEEN ?1 AND ?2",
            params![start, end],
            |row| row.get(0),
        )
        .unwrap_or(0.0);

    let avg_brightness: f64 = conn
        .query_row(
            "SELECT COALESCE(AVG(brightness), 0) FROM frames WHERE timestamp BETWEEN ?1 AND ?2",
            params![start, end],
            |row| row.get(0),
        )
        .unwrap_or(0.0);

    let mut stmt = conn
        .prepare(
            "SELECT CAST(strftime('%H', timestamp) AS INTEGER) as hour, COUNT(*) as count
             FROM frames WHERE timestamp BETWEEN ?1 AND ?2
             GROUP BY hour ORDER BY hour",
        )
        .map_err(|e| e.to_string())?;

    let hourly: HashMap<i64, i64> = stmt
        .query_map(params![start, end], |row| Ok((row.get(0)?, row.get(1)?)))
        .map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();

    let activity: Vec<i64> = (0..24).map(|h| *hourly.get(&h).unwrap_or(&0)).collect();

    Ok(DayStats {
        date,
        frames,
        events,
        summaries,
        avg_motion,
        avg_brightness,
        activity,
    })
}

#[tauri::command]
pub fn get_activities(date: String, db: State<AppDb>) -> Result<ActivityStats, String> {
    let conn = db.conn.lock().map_err(|e| e.to_string())?;
    let start = format!("{date}T00:00:00");
    let end = format!("{date}T23:59:59");

    // Estimate frame interval
    let frame_duration: i64 = conn
        .query_row(
            "SELECT MIN(julianday(t2.timestamp) - julianday(t1.timestamp)) * 86400
             FROM frames t1, frames t2
             WHERE t1.timestamp BETWEEN ?1 AND ?2 AND t2.timestamp BETWEEN ?1 AND ?2
               AND t2.rowid = t1.rowid + 1",
            params![start, end],
            |row| row.get::<_, Option<f64>>(0),
        )
        .unwrap_or(None)
        .map(|v| if v > 0.0 { v.round() as i64 } else { 30 })
        .unwrap_or(30);

    let mut stmt = conn
        .prepare(
            "SELECT activity, COUNT(*) as frame_count
             FROM frames WHERE timestamp BETWEEN ?1 AND ?2 AND activity != ''
             GROUP BY activity ORDER BY frame_count DESC",
        )
        .map_err(|e| e.to_string())?;

        .collect();

    let mut stmt2 = conn
        .prepare(

             FROM frames WHERE timestamp BETWEEN ?1 AND ?2 AND activity != ''

#[tauri::command]
pub fn get_apps(date: String, db: State<AppDb>) -> Result<Vec<AppStat>, String> {
    let conn = db.conn.lock().map_err(|e| e.to_string())?;
    let start = format!("{date}T00:00:00");
    let end = format!("{date}T23:59:59");

    // Check if window_events table exists
    let table_exists: bool = conn
        .query_row(
            "SELECT COUNT(*) FROM sqlite_master WHERE type='table' AND name='window_events'",
            [],
            |row| row.get::<_, i64>(0),
        )
        .unwrap_or(0)
        > 0;

    if !table_exists {
        return Ok(vec![]);
    }

    let mut stmt = conn
        .prepare(
            "SELECT timestamp, process_name, window_title,
                    LEAD(timestamp, 1, ?1) OVER (ORDER BY timestamp) as next_ts
             FROM window_events
             WHERE timestamp BETWEEN ?2 AND ?1
             ORDER BY timestamp",
        )
        .map_err(|e| e.to_string())?;

    struct WinRow {
        timestamp: String,
        process_name: String,
        window_title: String,
        next_ts: String,
    }

    let rows: Vec<WinRow> = stmt
        .query_map(params![end, start], |row| {
            Ok(WinRow {
                timestamp: row.get(0)?,
                process_name: row.get::<_, Option<String>>(1)?.unwrap_or_default(),
                window_title: row.get::<_, Option<String>>(2)?.unwrap_or_default(),
                next_ts: row.get(3)?,
            })
        })
        .map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();

    let mut by_process: HashMap<String, (f64, String, f64, i64)> = HashMap::new();

    for row in &rows {
        if row.process_name.is_empty() {
            continue;
        }
        let ts_ms = chrono::NaiveDateTime::parse_from_str(&row.timestamp, "%Y-%m-%dT%H:%M:%S")
            .or_else(|_| chrono::NaiveDateTime::parse_from_str(&row.timestamp, "%Y-%m-%dT%H:%M:%S%.f"));
        let next_ms = chrono::NaiveDateTime::parse_from_str(&row.next_ts, "%Y-%m-%dT%H:%M:%S")
            .or_else(|_| chrono::NaiveDateTime::parse_from_str(&row.next_ts, "%Y-%m-%dT%H:%M:%S%.f"));

        if let (Ok(ts), Ok(next)) = (ts_ms, next_ms) {
            let duration = (next - ts).num_milliseconds() as f64 / 1000.0;
            if duration <= 0.0 || duration > 86400.0 {
                continue;
            }

            let entry = by_process
                .entry(row.process_name.clone())
                .or_insert((0.0, String::new(), 0.0, 0));
            entry.0 += duration;
            entry.3 += 1;
            if duration > entry.2 {
                entry.1 = row.window_title.clone();
                entry.2 = duration;
            }
        }
    }

    let mut apps: Vec<AppStat> = by_process
        .into_iter()
        .map(|(process, (dur, title, _, switches))| AppStat {
            process,
            title_sample: title,
            duration_sec: dur.round() as i64,
            switch_count: switches,
        })
        .collect();

    apps.sort_by(|a, b| b.duration_sec.cmp(&a.duration_sec));
    Ok(apps)
}

#[tauri::command]
pub fn get_range_stats(from: String, to: String, db: State<AppDb>) -> Result<RangeStats, String> {
    let conn = db.conn.lock().map_err(|e| e.to_string())?;
    let start = format!("{from}T00:00:00");
    let end = format!("{to}T23:59:59");

    // Frame interval
    let frame_duration: i64 = conn
        .query_row(
            "SELECT MIN(julianday(t2.timestamp) - julianday(t1.timestamp)) * 86400
             FROM frames t1, frames t2
             WHERE t1.timestamp BETWEEN ?1 AND ?2 AND t2.timestamp BETWEEN ?1 AND ?2
               AND t2.rowid = t1.rowid + 1",
            params![start, end],
            |row| row.get::<_, Option<f64>>(0),
        )
        .unwrap_or(None)
        .map(|v| if v > 0.0 { v.round() as i64 } else { 30 })
        .unwrap_or(30);

    // Per-day frames
    let mut stmt = conn
        .prepare(
            "SELECT date(timestamp) as d, COUNT(*) as frame_count
             FROM frames WHERE timestamp BETWEEN ?1 AND ?2
             GROUP BY d ORDER BY d",
        )
        .map_err(|e| e.to_string())?;

    let daily_frames: Vec<(String, i64)> = stmt
        .query_map(params![start, end], |row| Ok((row.get(0)?, row.get(1)?)))
        .map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();

    // Per-day activities
    let mut stmt2 = conn
        .prepare(
            "SELECT date(timestamp) as d, activity, COUNT(*) as frame_count
             FROM frames WHERE timestamp BETWEEN ?1 AND ?2 AND activity != ''
             GROUP BY d, activity ORDER BY d, frame_count DESC",
        )
        .map_err(|e| e.to_string())?;

    let daily_activities: Vec<(String, String, i64)> = stmt2
        .query_map(params![start, end], |row| {
            Ok((row.get(0)?, row.get(1)?, row.get(2)?))
        })
        .map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();

    // Build days
    let mut activity_totals: HashMap<String, i64> = HashMap::new();
    let mut meta_totals: HashMap<String, i64> = HashMap::new();

    let days: Vec<RangeDay> = daily_frames
        .iter()
        .map(|(d, fc)| {
            let mut activities: HashMap<String, i64> = HashMap::new();
            let mut meta_cats: HashMap<String, i64> = HashMap::new();

            for (dd, act, afc) in &daily_activities {
                if dd == d {
                    let sec = afc * frame_duration;
                    activities.insert(act.clone(), sec);
                    *activity_totals.entry(act.clone()).or_insert(0) += sec;

                    let meta = db.get_meta_category(act);
                    *meta_cats.entry(meta.clone()).or_insert(0) += afc;
                    *meta_totals.entry(meta).or_insert(0) += afc;
                }
            }

            RangeDay {
                date: d.clone(),
                frame_count: *fc,
                total_sec: fc * frame_duration,
                activities,
                meta_categories: meta_cats,
            }
        })
        .collect();

    let total_frames: i64 = days.iter().map(|d| d.frame_count).sum();

    Ok(RangeStats {
        from,
        to,
        frame_duration,
        total_frames,
        total_sec: total_frames * frame_duration,
        days,
        activity_totals,
        meta_totals,
    })
}

use crate::db::AppDb;
use crate::models::Session;
use rusqlite::params;
use tauri::State;

#[tauri::command]
pub fn get_sessions(date: String, db: State<AppDb>) -> Result<Vec<Session>, String> {
    let conn = db.conn.lock().map_err(|e| e.to_string())?;
    let start = format!("{date}T00:00:00");
    let end = format!("{date}T23:59:59");

    let mut stmt = conn
        .prepare(
            "SELECT id, timestamp, activity FROM frames
             WHERE timestamp BETWEEN ?1 AND ?2 AND activity != ''
             ORDER BY timestamp",
        )
        .map_err(|e| e.to_string())?;

    let frames: Vec<(i64, String, String)> = stmt
        .query_map(params![start, end], |row| {
            Ok((row.get(0)?, row.get(1)?, row.get(2)?))
        })
        .map_err(|e| e.to_string())?
        .filter_map(|r| r.ok())
        .collect();

    if frames.is_empty() {
        return Ok(vec![]);
    }

    // Estimate interval
    let interval_sec: i64 = if frames.len() >= 2 {
        let t0 = chrono::NaiveDateTime::parse_from_str(&frames[0].1, "%Y-%m-%dT%H:%M:%S")
            .or_else(|_| chrono::NaiveDateTime::parse_from_str(&frames[0].1, "%Y-%m-%dT%H:%M:%S%.f"));
        let t1 = chrono::NaiveDateTime::parse_from_str(&frames[1].1, "%Y-%m-%dT%H:%M:%S")
            .or_else(|_| chrono::NaiveDateTime::parse_from_str(&frames[1].1, "%Y-%m-%dT%H:%M:%S%.f"));
        if let (Ok(a), Ok(b)) = (t0, t1) {
            let diff = (b - a).num_seconds();
            if diff > 0 && diff < 300 { diff } else { 30 }
        } else {
            30
        }
    } else {
        30
    };

    let mut sessions = Vec::new();
    let mut session_start = 0usize;
    let mut frame_count: i64 = 1;

    for i in 1..frames.len() {
        if frames[i].2 == frames[session_start].2 {
            frame_count += 1;
        } else {
            sessions.push(Session {
                activity: frames[session_start].2.clone(),
                meta_category: db.get_meta_category(&frames[session_start].2),
                start_time: frames[session_start].1.clone(),
                end_time: frames[i - 1].1.clone(),
                duration_sec: frame_count * interval_sec,
                frame_count,
            });
            session_start = i;
            frame_count = 1;
        }
    }

    // Last session
    sessions.push(Session {
        activity: frames[session_start].2.clone(),
        meta_category: db.get_meta_category(&frames[session_start].2),
        start_time: frames[session_start].1.clone(),
        end_time: frames.last().unwrap().1.clone(),
        duration_sec: frame_count * interval_sec,
        frame_count,
    });

    Ok(sessions)
}

let convertFileSrc: ((path: string) => string) | null = null;

// Lazy-load Tauri API. Falls back to /media/ prefix for non-Tauri environments.
async function loadTauri() {
  if (convertFileSrc) return;
  try {
    const { convertFileSrc: cfs } = await import('@tauri-apps/api/core');
    convertFileSrc = cfs;
  } catch {
    convertFileSrc = null;
  }
}

// Initialize on module load
loadTauri();

/**
 * Resolve a relative media path (e.g. "frames/2024-01-01/12-00-00.jpg")
 * to a URL usable in <img src> or <audio src>.
 *
 * In Tauri: asset://localhost/<absolute-path>
 * In browser: /media/<relative-path>
 */
export function mediaUrl(relativePath: string): string {
  if (!relativePath) return '';

  // If we have Tauri's convertFileSrc and know the data dir, use asset protocol
  if (convertFileSrc && (window as unknown as { __TAURI__?: unknown }).__TAURI__) {
    // The data_dir is injected by Tauri at startup
    const dataDir = (window as unknown as { __DATA_DIR__?: string }).__DATA_DIR__ || '';
    if (dataDir) {
      const sep = dataDir.includes('\\') ? '\\' : '/';
      return convertFileSrc(`${dataDir}${sep}${relativePath}`);
    }
  }

  // Fallback: serve via /media/ (Hono server or Vite proxy)
  return `/media/${relativePath}`;
}
